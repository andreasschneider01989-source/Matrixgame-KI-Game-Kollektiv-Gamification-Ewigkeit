const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');
let W, H;

  W = canvas.width  = window.innerWidth;
  H = canvas.height = window.innerHeight;

window.addEventListener('resize', resize);

// ── Physics Constants ──────────────────────────
const G    = 300;        // Gravitationskonstante (skaliert)
const DT   = 0.016;      // Zeitschritt
const MIN_DIST = 8;      // Mindestabstand für Kraftberechnung
const DECAY_PROB = 0.0005; // Zerfallswahrscheinlichkeit pro Tick

// ── State ──────────────────────────────────────
let bodies   = [];
let tick     = 0;
let paused   = false;
let zoom     = 1;
let panX     = 0;
let panY     = 0;
let mode     = 'gravity';
let collideCount = 0;
let decayCount   = 0;
let entropy  = 0.3;

// Colour palettes
const PALETTES = {
  gravity: ['#00ffb3','#00e5a0','#00cc8f','#60a5fa','#93c5fd'],
  repulse: ['#ff6b35','#ff8c42','#ffa726','#ffcc02','#ffe066'],
  chaos:   ['#f472b6','#e879f9','#c084fc','#818cf8','#60a5fa'],
  fractal: ['#7c3aed','#8b5cf6','#a78bfa','#c4b5fd','#ddd6fe'],
};

// ── Body Class ─────────────────────────────────
class Body {
  constructor(x, y, z, vx, vy, vz, mass, color, movable=true) {
    this.x = x; this.y = y; this.z = z;
    this.vx = vx; this.vy = vy; this.vz = vz;
    this.mass = mass;
    this.color = color;
    this.movable = movable;
    this.radius = Math.max(2, Math.cbrt(mass) * 1.5);
    this.trail = [];
    this.age = 0;
    this.alive = true;
  }

  // Screen projection: simple isometric-ish for 3D feel
  screenX() { return this.x * zoom + panX + W/2; }
  screenY() { return (this.y - this.z * 0.4) * zoom + panY + H/2; }

  addTrail() {
    this.trail.push({ x: this.screenX(), y: this.screenY() });
    if (this.trail.length > 30) this.trail.shift();
  }
}

// ── Initialise Simulation ──────────────────────
function initDefault() {
  bodies = [];
  collideCount = 0;
  decayCount   = 0;
  tick = 0;

  // Central star
  bodies.push(new Body(0, 0, 0,  0, 0, 0,  50000, '#ffe066', false));

  // Orbiting planets
  const pal = PALETTES['gravity'];
  const orbitData = [
    {r:120, m:200,  v:1.8},
    {r:200, m:80,   v:1.4},
    {r:290, m:350,  v:1.1},
    {r:380, m:50,   v:0.95},
    {r:460, m:150,  v:0.82},
  ];
  orbitData.forEach((od, i) => {
    const angle = Math.random() * Math.PI * 2;
    const z     = (Math.random()-0.5) * 40;
    const vOrbit = Math.sqrt(G * bodies[0].mass / od.r);
    bodies.push(new Body(
      Math.cos(angle)*od.r, Math.sin(angle)*od.r, z,
      -Math.sin(angle)*vOrbit, Math.cos(angle)*vOrbit, 0,
      od.m, pal[i % pal.length]
    ));
  });
}

// ── Spawn functions ────────────────────────────
function addBody() {
  const pal = PALETTES[mode];
  const angle = Math.random()*Math.PI*2;
  const r     = 100 + Math.random()*300;
  const z     = (Math.random()-0.5)*80;
  const star  = bodies.find(b => !b.movable) || bodies[0];
  const vO    = Math.sqrt(G * (star ? star.mass : 10000) / r) * (0.8 + Math.random()*0.4);
  const mass  = 10 + Math.random()*200;
  const c = pal[Math.floor(Math.random()*pal.length)];
  bodies.push(new Body(
    Math.cos(angle)*r, Math.sin(angle)*r, z,
    -Math.sin(angle)*vO, Math.cos(angle)*vO, (Math.random()-0.5)*0.5,
    mass, c
  ));
  logEvent(`+ Körper hinzugefügt (M=${Math.round(mass)})`, 'event-spawn');
}

function addCluster() {
  for(let i=0;i<8;i++) addBody();
  logEvent('+ Cluster (8 Körper) gespawnt', 'event-spawn');
}

function bigBang() {
  bodies = [];
  tick = 0;
  const n = 60 + Math.floor(Math.random()*40);
  const pal = PALETTES[mode];
  for(let i=0;i<n;i++) {
    const angle = Math.random()*Math.PI*2;
    const elev  = (Math.random()-0.5)*Math.PI;
    const r     = Math.random()*30;
    const speed = 50 + Math.random()*300;
    const mass  = 5 + Math.random()*500;
    bodies.push(new Body(
      Math.cos(angle)*r, Math.sin(angle)*r, Math.sin(elev)*r,
      Math.cos(angle)*speed*(Math.random()-0.5)*2,
      Math.sin(angle)*speed*(Math.random()-0.5)*2,
      (Math.random()-0.5)*speed,
      mass, pal[Math.floor(Math.random()*pal.length)]
    ));
  }
  logEvent(`⚡ BIG BANG! ${n} Körper`, 'event-collision');
}

function resetSim() {
  initDefault();
  zoom = 1; panX = 0; panY = 0;
  logEvent('⟳ Simulation zurückgesetzt', '');
}

// ── Physics Step ───────────────────────────────
function physicsStep() {
  const n = bodies.length;
  const ax = new Float64Array(n);
  const ay = new Float64Array(n);
  const az = new Float64Array(n);

  for(let i=0;i<n;i++) {
    const bi = bodies[i];
    if(!bi.movable || !bi.alive) continue;
    for(let j=0;j<n;j++) {
      if(i===j) continue;
      const bj = bodies[j];
      if(!bj.alive) continue;
      const dx = bj.x - bi.x;
      const dy = bj.y - bi.y;
      const dz = bj.z - bi.z;
      const r2 = Math.max(dx*dx + dy*dy + dz*dz, MIN_DIST*MIN_DIST);
      const r  = Math.sqrt(r2);

      // Collision check
      if(r < (bi.radius + bj.radius) * 0.8 && bi.mass > 1 && bj.mass > 1) {
        handleCollision(i, j);
        continue;
      }

      let force = 0;
      if(mode === 'gravity') {
        force = G * bj.mass / r2;
      } else if(mode === 'repulse') {
        force = -G * bj.mass / r2 * 0.5;
      } else if(mode === 'chaos') {
        force = G * bj.mass / r2 * (Math.random() > 0.5 ? 1 : -1);
      } else if(mode === 'fractal') {
        // Self-similar: force modulated by Mandelbrot-inspired factor
        const s = Math.sin(r * 0.05) * 0.5 + 0.5;
        force = G * bj.mass / r2 * (0.3 + s * 1.4);
      }

      ax[i] += force * dx / r;
      ay[i] += force * dy / r;
      az[i] += force * dz / r;
    }
  }

  let totalKE = 0;
  for(let i=0;i<n;i++) {
    const b = bodies[i];
    if(!b.alive) continue;
    if(b.movable) {
      b.vx += ax[i] * DT;
      b.vy += ay[i] * DT;
      b.vz += az[i] * DT;
      // Velocity cap to prevent instability
      const speed = Math.sqrt(b.vx*b.vx + b.vy*b.vy + b.vz*b.vz);
      if(speed > 800) { const s=800/speed; b.vx*=s; b.vy*=s; b.vz*=s; }
      b.x += b.vx * DT;
      b.y += b.vy * DT;
      b.z += b.vz * DT;
    }
    b.addTrail();
    b.age++;
    totalKE += 0.5 * b.mass * (b.vx*b.vx + b.vy*b.vy + b.vz*b.vz);

    // Random decay (Formelsammlung: stochastic decay event)
    if(b.movable && Math.random() < DECAY_PROB && b.mass > 50) {
      decay(i);
    }
  }

  // Clean dead bodies
  bodies = bodies.filter(b => b.alive);

  // Update entropy (normalised KE variance)
  const speeds = bodies.map(b => Math.sqrt(b.vx*b.vx+b.vy*b.vy+b.vz*b.vz));
  const mean = speeds.reduce((a,v)=>a+v,0) / (speeds.length||1);
  const variance = speeds.reduce((a,v)=>a+(v-mean)*(v-mean),0) / (speeds.length||1);
  entropy = Math.min(1, Math.sqrt(variance) / 200);

  // Update Bayesian confidences dynamically
  updateBayes(totalKE);

  tick++;
}

// ── Collision ──────────────────────────────────
const collisionSet = new Set();
function handleCollision(i, j) {
  const key = Math.min(i,j)+'-'+Math.max(i,j);
  if(collisionSet.has(key)) return;
  collisionSet.add(key);
  setTimeout(()=>collisionSet.delete(key), 200);

  const bi = bodies[i], bj = bodies[j];
  if(!bi || !bj || !bi.alive || !bj.alive) return;

  // Perfectly inelastic merge if one is big
  if(bi.mass > 500 || bj.mass > 500) {
    const total = bi.mass + bj.mass;
    if(!bi.movable) { bj.alive=false; }
    else if(!bj.movable) { bi.alive=false; }
    else {
      bi.vx=(bi.vx*bi.mass+bj.vx*bj.mass)/total;
      bi.vy=(bi.vy*bi.mass+bj.vy*bj.mass)/total;
      bi.vz=(bi.vz*bi.mass+bj.vz*bj.mass)/total;
      bi.mass=total;
      bi.radius=Math.max(3, Math.cbrt(total)*1.5);
      bj.alive=false;
    }
    collideCount++;
    logEvent(`💥 Kollision! Verschmelzung`, 'event-collision');
  }
}

// ── Decay ──────────────────────────────────────
function decay(i) {
  const b = bodies[i];
  if(!b || !b.alive) return;
  const pal = PALETTES[mode];
  const fragments = Math.min(5, Math.floor(b.mass/20));
  for(let k=0;k<fragments;k++) {
    const c = pal[Math.floor(Math.random()*pal.length)];
    const frag = new Body(
      b.x+(Math.random()-0.5)*20,
      b.y+(Math.random()-0.5)*20,
      b.z+(Math.random()-0.5)*20,
      b.vx+(Math.random()-0.5)*30,
      b.vy+(Math.random()-0.5)*30,
      b.vz+(Math.random()-0.5)*10,
      b.mass/fragments, c
    );
    bodies.push(frag);
  }
  b.alive = false;
  decayCount++;
  logEvent(`☢ Zerfall! → ${fragments} Fragmente`, 'event-decay');
}

// ── Render ─────────────────────────────────────
function render() {
  ctx.clearRect(0, 0, W, H);

  // Sort by z for painter's algorithm
  const sorted = [...bodies].sort((a,b) => a.z - b.z);

  for(const b of sorted) {
    if(!b.alive) continue;
    const sx = b.screenX();
    const sy = b.screenY();

    // Draw trail
    if(b.trail.length > 1) {
      ctx.beginPath();
      ctx.moveTo(b.trail[0].x, b.trail[0].y);
      for(let k=1;k<b.trail.length;k++) {
        ctx.lineTo(b.trail[k].x, b.trail[k].y);
      }
      ctx.strokeStyle = b.color + '40';
      ctx.lineWidth = 1;
      ctx.stroke();
    }

    // 3D depth cue: z affects brightness
    const depthFactor = 0.7 + (b.z / 200 + 0.5) * 0.3;
    const r = Math.max(2, b.radius * zoom * depthFactor);

    // Glow
    const grd = ctx.createRadialGradient(sx, sy, 0, sx, sy, r*3);
    grd.addColorStop(0, b.color + 'cc');
    grd.addColorStop(0.4, b.color + '55');
    grd.addColorStop(1, 'transparent');
    ctx.beginPath();
    ctx.arc(sx, sy, r*3, 0, Math.PI*2);
    ctx.fillStyle = grd;
    ctx.fill();

    // Core
    ctx.beginPath();
    ctx.arc(sx, sy, r, 0, Math.PI*2);
    ctx.fillStyle = b.color;
    ctx.fill();

    // White specular highlight
    const hl = ctx.createRadialGradient(sx-r*0.3, sy-r*0.3, 0, sx, sy, r);
    hl.addColorStop(0, 'rgba(255,255,255,0.4)');
    hl.addColorStop(1, 'transparent');
    ctx.beginPath();
    ctx.arc(sx, sy, r, 0, Math.PI*2);
    ctx.fillStyle = hl;
    ctx.fill();
  }

  // Gravity field lines (when in gravity mode)
  if(mode === 'gravity' && bodies.length > 0) {
    const star = bodies.find(b => !b.movable);
    if(star) {
      const sx = star.screenX(), sy = star.screenY();
      for(let angle=0; angle<Math.PI*2; angle+=Math.PI/6) {
        ctx.beginPath();
        ctx.moveTo(sx, sy);
        ctx.lineTo(sx + Math.cos(angle)*500*zoom, sy + Math.sin(angle)*500*zoom);
        ctx.strokeStyle = 'rgba(255,230,2,0.03)';
        ctx.lineWidth = 1;
        ctx.stroke();
      }
    }
  }
}

let lastBayesUpdate = 0;
const bayesPct = [85, 78, 99, 99, 42];

function updateBayes(ke) {
  if(tick - lastBayesUpdate < 60) return;
  lastBayesUpdate = tick;

  // Bayesian updates based on simulation events
  if(collideCount > 5) {
    bayesPct[0] = Math.min(99, bayesPct[0] + 0.5); // more evidence for entropy
  }
  if(bodies.length > 20) {
    bayesPct[1] = Math.min(98, bayesPct[1] + 0.3); // fractal structures
  }
  // Energy conservation fluctuates in chaos mode
  if(mode === 'chaos') {
    bayesPct[2] = Math.max(70, bayesPct[2] - 0.2);
  } else {
    bayesPct[2] = Math.min(99, bayesPct[2] + 0.05);
  }

  const ids = ['b0','b1','b2','b3','b4'];
  const pids = ['bp0','bp1','bp2','bp3','bp4'];
  ids.forEach((id,i) => {
    document.getElementById(id).style.width = bayesPct[i]+'%';
    document.getElementById(pids[i]).textContent = Math.round(bayesPct[i])+'%';
  });
}

  if(dragging) { panX+=e.clientX-lastMX; panY+=e.clientY-lastMY; lastMX=e.clientX; lastMY=e.clientY; }

}, {passive:false});

// Touch support
let lastTouchDist = null;
canvas.addEventListener('touchstart', e => {
  if(e.touches.length===1) { dragging=true; lastMX=e.touches[0].clientX; lastMY=e.touches[0].clientY; }
  if(e.touches.length===2) { const dx=e.touches[0].clientX-e.touches[1].clientX; const dy=e.touches[0].clientY-e.touches[1].clientY; lastTouchDist=Math.sqrt(dx*dx+dy*dy); }
});
canvas.addEventListener('touchmove', e => {
  e.preventDefault();
  if(e.touches.length===1 && dragging) { panX+=e.touches[0].clientX-lastMX; panY+=e.touches[0].clientY-lastMY; lastMX=e.touches[0].clientX; lastMY=e.touches[0].clientY; }
  if(e.touches.length===2 && lastTouchDist) { const dx=e.touches[0].clientX-e.touches[1].clientX; const dy=e.touches[0].clientY-e.touches[1].clientY; const d=Math.sqrt(dx*dx+dy*dy); if(d>lastTouchDist) zoomIn(); else zoomOut(); lastTouchDist=d; }
}, {passive:false});
canvas.addEventListener('touchend', ()=>{ dragging=false; lastTouchDist=null; });
