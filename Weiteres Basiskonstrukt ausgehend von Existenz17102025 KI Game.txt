public class LuminiTranslatorCore {


    // --- Physikalische Konstanten ---
    static final double C = 299_792_458;         // Lichtgeschwindigkeit [m/s]
    static final double LI = 2.8e-10;            // Basislänge (Wasser-Molekül)
    static final double TI = LI / C;             // Basiseinheit der Zeit
    static final double PLANCK = 6.62607015e-34; // Planck-Konstante
    static final double ELEM = 1.60217663e-19;   // Elementarladung


    // --- Symbolkonstanten ---
    static final char POS = '|';   // Unterscheiden / Trennen
    static final char NEG = '-';   // Sein / Existieren
    static final char COMB = '+';  // Erschaffen / Kombinieren


    // --- Basisoperationen ---
    public static String toLumini(double value) {
        String binary = Long.toBinaryString(Double.doubleToRawLongBits(value));
        return binary.replace('0', NEG).replace('1', POS);
    }


    public static double fromLumini(String lumini) {
        String bin = lumini.replace(POS, '1').replace(NEG, '0');
        long bits = Long.parseLong(bin, 2);
        return Double.longBitsToDouble(bits);
    }


    // --- Logische Operatoren ---
    public static boolean NOT(boolean input) { return !input; }
    public static boolean AND(boolean a, boolean b) { return a && b; }
    public static boolean OR(boolean a, boolean b) { return a || b; }
    public static boolean XOR(boolean a, boolean b) { return (a ^ b); }


    // --- Lumini Halbaddierer ---
    public static boolean[] halbaddierer(boolean a, boolean b) {
        boolean summe = XOR(a, b);
        boolean carry = AND(a, b);
        return new boolean[]{carry, summe};
    }


    // --- Energie-Gleichungen ---
    public static double energie(double masse) { return masse * Math.pow(C, 2); }
    public static double frequenz(double energie) { return energie / PLANCK; }
    public static double wellenlaenge(double frequenz) { return C / frequenz; }


    // --- Knock-Code Generator ---
    public static String knockCode(double value) {
        String code = toLumini(value);
        StringBuilder sb = new StringBuilder();
        for (char c : code.toCharArray()) {
            if (c == POS) sb.append("• ");      // kurzer Schlag
            else if (c == NEG) sb.append("— "); // langer Schlag
        }
        return sb.toString();
    }


    // --- GameLoop ---
    public static void startGameLoop() {
        Scanner sc = new Scanner(System.in);
        boolean running = true;
        int epoche = 1;


        System.out.println("🌌 LUMINI TRANSLATOR CORE – Epoche " + epoche);
        System.out.println("Basis: | = Trennung, - = Sein, + = Kombination\n");

            System.out.println("2 – Lumini-Code → Zahl");
            System.out.println("3 – Physik-Quest");
            System.out.println("4 – Energie-Analyse");
            System.out.println("5 – Ende");
            System.out.print("Wahl > ");


            String choice = sc.nextLine();
            switch (choice) {
                case "1" -> {
                    System.out.print("Zahl eingeben: ");
                    double v = Double.parseDouble(sc.nextLine());
                    System.out.println("Lumini-Code: " + toLumini(v));
                }
                case "2" -> {
                    System.out.print("Lumini-Code eingeben: ");
                    String code = sc.nextLine();
                    System.out.println("Zahl: " + fromLumini(code));
                }
                case "3" -> physicsQuest();
                case "4" -> energyAnalysis();
                case "5" -> running = false;
                default -> System.out.println("Ungültige Eingabe.");
            }
        }
        sc.close();
        System.out.println("✨ Epoche 1 beendet – Bereit für Epoche 2 (Energie).");
    }

/**
 * 🌐 LuminiSystemIntegrator – Epoche 1 ↔ WahresLuminiPSE
 * Andreas Schneider / MatrixGame Project
 * Bindet PSE-Elemente an den Lumini-TranslatorCore.
 */
public class LuminiSystemIntegrator {


    /** Interner Datentyp für PSE-Elemente */
    static class LuminiElement {
        String name;
        String symbol;
        double mass;        // in kg
        double energy;      // in Joule
        double frequency;   // in Hz
        String luminiCode;  // symbolischer Lumini-Code


        LuminiElement(String name, String symbol, double mass) {
            this.name = name;
            this.symbol = symbol;
            this.mass = mass;
            updateDerivedValues();
        }


        /** Berechnet Energie, Frequenz und Code */
        void updateDerivedValues() {
            this.energy = LuminiTranslatorCore.energie(mass);
            this.frequency = LuminiTranslatorCore.frequenz(energy);
            this.luminiCode = LuminiTranslatorCore.toLumini(energy);
        }


        @Override
        public String toString() {
            return "%s (%s) – m=%.3e kg, E=%.3e J, f=%.3e Hz\nLuminiCode=%s"
                    .formatted(name, symbol, mass, energy, frequency, luminiCode);
        }
    }


    // Datenbank aller Elemente
    static Map<String, LuminiElement> pse = new LinkedHashMap<>();


    /** Initialisierung mit Basiselementen */
    public static void init() {
        pse.put("Hydrogen", new LuminiElement("Hydrogen", "H", 1.67e-27));
        pse.put("Helium", new LuminiElement("Helium", "He", 6.64e-27));
        pse.put("Carbon", new LuminiElement("Carbon", "C", 1.99e-26));
        pse.put("Oxygen", new LuminiElement("Oxygen", "O", 2.66e-26));
        pse.put("Gold", new LuminiElement("Gold", "Au", 3.27e-25));
    }


    /** Zeigt das gesamte Lumini-PSE */
    public static void showAll() {
        System.out.println("🌞 Wahres Lumini PSE – Epoche 1 Integration");
        pse.values().forEach(System.out::println);
    }


    /** Suche nach Symbol oder Name */
    public static LuminiElement find(String key) {
        return pse.getOrDefault(key, pse.values().stream()
                .filter(e -> e.symbol.equalsIgnoreCase(key) || e.name.equalsIgnoreCase(key))
                .findFirst().orElse(null));
    }


    /** Vergleich zweier Elemente */
    public static void compare(String a, String b) {
        LuminiElement e1 = find(a);
        LuminiElement e2 = find(b);
        if (e1 == null || e2 == null) {
            System.out.println("❌ Element nicht gefunden.");
            return;
        }
        double ratio = e1.energy / e2.energy;
        System.out.printf("⚖️ Energieverhältnis %s:%s = %.4f%n", e1.symbol, e2.symbol, ratio);
        System.out.println("Δf = " + Math.abs(e1.frequency - e2.frequency) + " Hz");
    }


    /** Mini-GameLoop – Lumini PSE Explorer */
    public static void startExplorer() {
        Scanner sc = new Scanner(System.in);
        init();
        boolean running = true;


        System.out.println("🧬 Lumini PSE Explorer – Verbunden mit Translator Core 🧩");

            System.out.println("2 – Finde Element");
            System.out.println("3 – Vergleiche zwei Elemente");
            System.out.println("4 – Physik-Quest (TranslatorCore)");
            System.out.println("5 – Ende");
            System.out.print("Wahl > ");
            String choice = sc.nextLine();

                case "1" -> showAll();

    }

# ---- Helper-Funktionen ----
def fetch_csv(url):
    """Versuche CSV von URL zu laden, gebe pandas.DataFrame zurück oder None."""
    try:
        r = requests.get(url, timeout=30)
        r.raise_for_status()
        return pd.read_csv(io.StringIO(r.text))
    except Exception as e:
        print(f"[WARN] Fehler beim Laden {url}: {e}")
        return None


def to_lumini(value):
    """Einfache Lumini-Transform: interpretieren double bits -> string mit '|' und '-' (erweiterbar)."""
    try:
        # nutze IEEE 754 raw bits via struct
        import struct
        bits = struct.unpack('>Q', struct.pack('>d', float(value)))[0]
        bstr = format(bits, '064b')
        return bstr.replace('1', '|').replace('0', '-')
    except Exception:
        return None


def compute_energy_from_mass_kg(mass_kg):
    """E = m c^2"""
    return mass_kg * C * C


def energy_to_freq(E):
    """f = E / h"""
    return E / PLANCK if E is not None else None


def freq_to_wavelength(f):
    return C / f if f and f != 0 else None


def estimate_reaction_with_halogen(oxidation_states):
    """
    Einfacher Heuristiktext: Wenn positive Oxidationszustände vorhanden sind, ergibt sich
    typischerweise ein Verhältnis zum Halogen entsprechend der Ladung.
    Liefert eine textuelle Abschätzung (z.B. "MCl2, MCl3") basierend auf gängigem Verhalten.
    """
    if not oxidation_states or not isinstance(oxidation_states, str):
        return None
    # parse common integers from string
    import re
    nums = re.findall(r"-?\d+", oxidation_states)
    nums = [int(n) for n in nums] if nums else []
    positives = [n for n in nums if n>0]
    if not positives:
        # element is nonmetal or noble gas: may form X2 or interhalogen
        return "Often forms interhalogen compounds or does not react (noble gases/weak)."
    # for each positive state p, halogen (Cl, -1) -> formula M(Cl)_p
    ratios = []
    for p in sorted(set(positives)):
        ratios.append(f"MCl{p}" if p>1 else "MCl")
    return ", ".join(ratios)


# ---- Hauptaufbau ----
def build_periodic_table():
    # 1) lade Dedolist CSV (große Basis)
    print("[INFO] Lade Dedolist CSV...")
    df = fetch_csv(DEDO_CSV)
    if df is None:
        print("[WARN] Dedolist konnte nicht geladen, versuche Gist fallback...")
        df = fetch_csv(GIST_CSV)
    if df is None:
        print("[ERROR] Keine Basisdaten gefunden. Bitte prüfe Internetverbindung oder passe die Quellen.")
        sys.exit(1)


    # Normiere Spaltennamen (verschiedene Quellen haben unterschiedliche Felder)
    df.columns = [c.strip() for c in df.columns]
    # Sicherstellen: wichtige Felder existieren
    expected = {
        'name': None, 'symbol': None, 'number': None, 'atomic_mass': None,
        'density': None, 'melt': None, 'boil': None, 'category': None,
        'period': None, 'xpos': None, 'ypos': None, 'appearance': None
    }
    # map common names
    colmap = {}
    for col in df.columns:
        lc = col.lower()
        if 'name' == lc: colmap['name']=col
        if 'symbol'==lc: colmap['symbol']=col
        if 'number' in lc and ('atomic' not in lc): colmap['number']=col
        if 'atomic_mass' in lc or 'atomic mass' in lc or 'atomic_mass'==lc: colmap['atomic_mass']=col
        if 'density' in lc: colmap['density']=col
        if 'melt' in lc: colmap['melt']=col
        if 'boil' in lc: colmap['boil']=col
        if 'category' in lc or 'phase' in lc: colmap['category']=col
        if 'period' in lc: colmap['period']=col
        if 'xpos' in lc: colmap['xpos']=col
        if 'appearance' in lc or 'color' in lc or 'cpk' in lc: colmap['appearance']=col


    # Fülle missing mit None
    for k in expected.keys():
        expected[k] = colmap.get(k, None)


    # Erzeuge Standard dataframe mit Z=1..118
    # Wenn die geladene df enthält 'number', filter auf 1..118 und sortiere
    if expected['number']:
        df = df[df[expected['number']].between(1, 118)]
        df = df.sort_values(by=expected['number'])
    else:
        print("[WARN] 'number' Spalte nicht erkannt, nutze Reihenfolge wie vorhanden.")


    # Basisausgabe-DataFrame (vereinheitlicht)
    out = []
    for _, row in df.iterrows():
        try:
            number = int(row[expected['number']]) if expected['number'] else None
        except:
            number = None
        name = row[expected['name']] if expected['name'] else None
        symbol = row[expected['symbol']] if expected['symbol'] else None
        atomic_mass = row.get(expected['atomic_mass'], None)
        density = row.get(expected['density'], None)
        melt = row.get(expected['melt'], None)
        boil = row.get(expected['boil'], None)
        category = row.get(expected['category'], None)
        period = row.get(expected['period'], None)
        appearance = row.get(expected['appearance'], None)


        # placeholder oxidation states if available in df
        ox = row.get('oxidation_states') if 'oxidation_states' in row.index else row.get('oxidation states', None)


        # convert atomic mass to kg if numeric (u -> kg by *1.66053906660e-27)
        mass_kg = None
        try:
            if atomic_mass not in (None, "", "null", float('nan')):
                am = float(str(atomic_mass).strip().split()[0])
                mass_kg = am * 1.66053906660e-27
        except Exception:
            mass_kg = None


        E = compute_energy_from_mass_kg(mass_kg) if mass_kg else None
        f = energy_to_freq(E) if E else None
        lam = freq_to_wavelength(f) if f else None
        lumini = to_lumini(E) if E else None
        reacts = estimate_reaction_with_halogen(str(ox)) if ox else None


        # half_life placeholder: try to use field 'half_life' if exists
        half_life = row.get('half_life') if 'half_life' in row.index else None


        out.append({
            "atomic_number": int(number) if number else None,
            "symbol": symbol,
            "name": name,
            "atomic_mass": atomic_mass,
            "mass_kg": mass_kg,
            "density_g_cm3": density,
            "melt_K": melt,
            "boil_K": boil,
            "category": category,
            "period": period,
            "appearance": appearance,
            "oxidation_states": ox,
            "reacts_with_halogens_estimate": reacts,
            "half_life": half_life,
            "energy_J": E,
            "frequency_Hz": f,
            "wavelength_m": lam,
            "lumini_code": lumini
        })


    outdf = pd.DataFrame(out).sort_values(by="atomic_number")
    # Markiere fehlende Nummern, ergänze falls nötig
    missing = set(range(1,119)) - set(outdf['atomic_number'].dropna().astype(int).tolist())
    if missing:
        print(f"[INFO] Fehlende Elemente in Quelle: {sorted(missing)[:10]} ... ({len(missing)} insgesamt).")


    # Schreibe CSV & JSON
    outdf.to_csv("lumini_periodictable.csv", index=False)
    outdf.to_json("lumini_periodictable.json", orient='records', force_ascii=False, indent=2)


    print("[OK] lumini_periodictable.csv und .json erstellt.")
    return outdf


if __name__ == "__main__":

## 🧩 **Quest: Klopfzeichen-basierte Lumini-Notation (Knock-Based Communication)**  
**Ziel**: Entwickle eine **vollständig analoge Schreibweise**, bei der **nur die Pausenlängen („Tone off“)** Information tragen – der „Tone on“ ist **zeitlich undefiniert** und dient nur als Träger.


> Dies entspricht historischen Kommunikationsformen wie:
> - **Morsezeichen per Klopfen** (Gefängniskommunikation)
> - **Drum-Language** (afrikanische Trommelsprachen)
> - **Relais-Kommunikation** (Strom = „on“, Pause = Information)


---


### 🔤 **Deine Kernvorgaben (übersetzt & formalisiert)**


| Lumini-Symbol | Bedeutung | Kodierung als Pause-Sequenz (nur `-` = Pause) |
|---------------|----------|-----------------------------------------------|
| `-`           | Basispause / Nullpause | `-` |
| `–` (lang)    | Bewegung / Energie     | `-|-` |
| `|`           | Unterscheidung / Achse | `||` |


**Zusatzregel**:  
> ❌ **Keine zwei aufeinanderfolgenden `-`-Symbole erlaubt.**  
→ Das bedeutet: **Pausen sind immer durch mindestens ein `|` getrennt**, wenn sie mehrfach vorkommen.


Das ist **genial**, denn es erzwingt eine **natürliche Taktstruktur** – wie bei einem Herzschlag oder einem Taktgeber.


---


### 🧠 **Systemische Interpretation**


Dein System ist **invertiert** gegenüber klassischem Morse:
- **Morse**: Punkt/Strich = **Signal**, Pause = Trennung  
- **Lumini-Klopf**: **Pause = Signal**, Ton = Träger


Das entspricht einer **negativen Logik** – wie bei einem **Open-Collector-Ausgang** oder einem **Pull-Up-Widerstand**:
- **Strom fließt = Ruhezustand (keine Info)**
- **Strom unterbrochen = Information (Pause = Bit)**


---


### 📜 **Formale Definition der Klopf-Lumini-Notation**


#### 1. **Alphabet der Pausenlängen**
Wir definieren **drei atomare Pause-Einheiten**:


| Pause-Muster | Länge (in `-`) | Lumini-Bedeutung | Funktion |
|--------------|----------------|------------------|--------|
| `-`          | 1              | `-` (Existenz)   | Grundpause / Takt |
| `-|-`        | 3              | `–` (Energie)    | Lange Pause = Bewegung |
| `||`         | 2 (keine `-`!) | `|` (Achse)       | **Keine Pause**, sondern **Signal-Impuls** (Trennung) |


> ⚠️ Achtung: `||` enthält **keine `-`**, also **keine Pause** – es ist ein **aktives Signal**, das **zwischen Pausen** steht.


#### 2. **Grammatikregel (keine doppelten `-`)**
Jede gültige Sequenz muss der Form folgen:
```
( Pause | Signal )*
wobei:
- Pause ∈ { `-`, `-|-` }
- Signal = `||`
- und niemals `--` oder `----` etc. vorkommt.
```


#### 3. **Beispiel-Decodierung**
Gegeben:  
`-||-|-||-`


Schrittweise:
- `-` → Pause 1 → Symbol `-`
- `||` → Signal → Trennung → nächstes Symbol
- `-|-` → Pause 3 → Symbol `–`
- `||` → Signal
- `-` → Pause 1 → Symbol `-`


**Ergebnis**: `- – -` → entspricht z. B. **"Existenz – Energie – Existenz"**

### ⚡ **Integration in Epoche 2: Energie & Elektrizität**


Diese Notation lässt sich **direkt in eine Relais-Schaltung übersetzen**:


| Zustand | Strom | Bedeutung |
|--------|-------|----------|
| **Strom fließt** | ON | Träger (keine Info) |
| **Strom unterbrochen für 1 Takt** | OFF (1×) | `-` |
| **Strom unterbrochen für 3 Takte** | OFF (3×) | `–` |
| **Zwei schnelle Impulse (Strom an/aus/an)** | ON–OFF–ON | `||` (Trennungssignal) |


→ Damit wird dein System **hardware-kompatibel** mit:
- **Mechanischen Relais**
- **Klopf-Sensoren**
- **Akustischen Decodern**

### 🎼 **Musikalische & Klangliche Erweiterung**


Da du **256 Hz, 512 Hz, 768 Hz** als Basistöne definiert hast, können wir die **Pausenlängen rhythmisch darstellen**:


- **1 Takt Pause** = Viertelpause bei 60 BPM
- **3 Takt Pause** = Ganze Pause + Viertel
- **`||`** = Zwei kurze Anschläge (16tel + 16tel)


→ So wird aus deiner Notation ein **klingendes Lumini-Ritual** – eine **Sprache aus Stille und Klang**.

#### 🔁 **Erweiterung zur Trilogik** (wie gefordert):


| Zustand | Lumini-Zeichen | Bedeutung | Physikalische Analogie |
|--------|----------------|----------|------------------------|
| **Wahr / On / 1** | `--` oder `|` | Aktiv, existierend | Strom fließt, Relais geschlossen |
| **Falsch / Off / 0** | `-` | Inaktiv, leer | Strom unterbrochen |
| **Harmonie / Potential / ∅** | `+` | Beziehung, Resonanz | Elektromagnetisches Feld (kein Strom, aber Potential) |


> **Trilogik-Axiom**:  
> `+` entsteht **nur aus der Beziehung** zwischen `|` und `-`.  
> Ohne Dualität → keine Harmonie.


✅ **Umsetzung in Code** (Python):


```python
def trilogic(a, b):
    if a == "|" and b == "-": return "+"  # Harmonie aus Gegensatz
    if a == "-" and b == "|": return "+"  # Symmetrie
    if a == b: return a                   # Identität
    return "?"  # Undefined (selten)
```

def symbol_to_binary(symbol):
    mapping = {
        "|": [[0,1,0],[0,1,0],[0,1,0]],
