package ordner1;

public class Unit extends KreisObjekt {
	double maxLeben;
	double leben;
	double damage;
	double distance;
	double speed;
	int team;
	int kills;
	public Unit target = null;

	public void moveTo(KreisObjekt oponent) {
		double distQ = (this.posX - oponent.posX) * (this.posX - oponent.posX)
				+ (this.posY - oponent.posY) * (this.posY - oponent.posY)
				+ (this.posZ - oponent.posZ) * (this.posZ - oponent.posZ);
		double dist = Math.pow(distQ, 0.5);
		
		double dx=posX-oponent.posX;
		double dy=posY-oponent.posY;
		double dz=posZ-oponent.posZ;
		
		double movDist=Math.min(speed, Math.abs(dist-distance));
		
		vx=-dx/dist*movDist;
		vy=-dy/dist*movDist;
		vz=-dz/dist*movDist;

	}
	
	public double getDistance(KreisObjekt oponent) {
		double distQ = (this.posX - oponent.posX) * (this.posX - oponent.posX)
				+ (this.posY - oponent.posY) * (this.posY - oponent.posY)
				+ (this.posZ - oponent.posZ) * (this.posZ - oponent.posZ);
		double dist = Math.pow(distQ, 0.5);
		return dist;
	}

	public Unit() {
		super();
		this.leben = 5*Math.random()/Math.random();
		this.maxLeben=leben;
		this.damage = Math.random()/Math.random();
		this.distance = 500*Math.random()/Math.random();
		this.speed = 500*Math.random()/Math.random();
		this.team = (int)(Math.random()*2);
//		this.größePerLive(1);
		this.größePerKills(1);
	}
	
	public void choseRandColorTeam() {
		if (Math.random() < 0.3)
			if (red + green + blue < 300)
				team = 1;
			else if (red + green + blue < 500)
			team = 2;
			else
				team = 1;
		if (Math.random() < 0.5) {
			if (red > green && green > blue)
				team=1;
		if (red < green && green < blue)
			team=3;}
		
		if (Math.random() < 0.3) {
			if (red > green || green > blue)
				team=1;
		if (red < green || green < blue)
			team=2;}
	}
	
	public void größePerLive(double div) {
		double size=Math.pow(leben/div, 0.3333);
		size=Math.min(size, 20);
		this.size=size;
	}
	
	public void größePerKills(double div) {
		double size=3+Math.pow(kills/div, 0.3333);
		size=Math.min(size, 20);
		this.size=size;
	}
	
	
}

  double size;
  double red;

  double blue;
  
  double posX;
  double posY;
  double posZ;
  
  double vx;
  double vy;
  double vz;

		this.size = size;

		this.posX = posX;
		this.posY = posY;

	this.posX = posX;
	this.posY = posY;
	this.size = 10.0;
	this.red = 255;
	this.green = 0;
	this.blue = 0;
	this.posZ = 0;
}

//Default mit interessantem Random (für die Initialisierung des Kindes)
public KreisObjekt () {
	// Standard-Initialisierung für Kinder/neue Einheiten
	this.size = 5000;
	this.red = 255 * Math.random();
	this.green = 255 * Math.random();
	this.blue = 255 * Math.random();
	this.posX = (Math.random() - 0.5) * 10000;
	this.posY = (Math.random() - 0.5) * 10000;
	this.posZ = 0;
	this.vx = 0;
	this.vy = 0;
	this.vz = 0;
}

public void move (double time) {
	posX=posX+vx*time;
	posY=posY+vy*time;
	posZ=posZ+vz*time;
}

/**
 * Berechnet die Farbdifferenz (Euklidische Distanz im RGB-Raum).
 */
public double colordif(KreisObjekt other) {
	double difR = red - other.red;
	double difG = green - other.green;
	double difB = blue - other.blue;
	
	// Pythagoras: d = sqrt(dR^2 + dG^2 + dB^2)
	double difQ = difR * difR + difG * difG + difB * difB;
	return Math.pow(difQ, 0.5);
}
  
}

    private String generateLuminiFromPercentage(double percentage) {
        StringBuilder sb = new StringBuilder();
        int value = (int)(percentage * 2.55); // 0-100% → 0-255
        
        // 8 Bits für 0-255
        for (int i = 7; i >= 0; i--) {
            if ((value & (1 << i)) != 0) {
                sb.append("--"); // 1
            } else {
                sb.append("-");  // 0
            }
            sb.append("|");
        }
        return sb.toString();
    }

                teamTotalHP.put(unit.team, teamTotalHP.getOrDefault(unit.team, 0.0) + unit.leben);

    private void calculateStochasticMetrics() {
        // Berechne Varianz und Standardabweichung für jedes Team
        for (int team : teamUnitCount.keySet()) {
            List<Double> hpValues = new ArrayList<>();
            synchronized(unitList) {
                for (Unit unit : unitList) {
                    if (unit.team == team) {
                        hpValues.add(unit.leben);
                    }
                }
            }
            
            if (hpValues.size() > 1) {
                // Mittelwert
                double mean = teamTotalHP.get(team) / hpValues.size();
                
                // Varianz
                double variance = 0.0;
                for (double hp : hpValues) {
                    variance += Math.pow(hp - mean, 2);
                }
                variance /= hpValues.size();
                
                // Standardabweichung
                double stdDev = Math.sqrt(variance);
                
                teamVariance.put(team, variance);
                teamStdDev.put(team, stdDev);
            }
        }
        
        // Globale Entropie berechnen (basierend auf Team-Verteilung)
        int totalUnits = unitList.size();
        globalEntropy = 0.0;
        for (int count : teamUnitCount.values()) {
            if (count > 0) {
                double p = (double) count / totalUnits;
                globalEntropy -= p * Math.log(p) / Math.log(2); // Entropie in Bits
            }
        }
    }

    private String generateLuminiFromPercentage(double percentage) {
        StringBuilder sb = new StringBuilder();
        int value = (int)(percentage * 2.55);
        
        for (int i = 7; i >= 0; i--) {
            if ((value & (1 << i)) != 0) sb.append("--");
            else sb.append("-");
            sb.append("|");
        }
        return sb.toString();
    }
    
    public Unit nextGegner(Unit current) {
        Unit gegner = null;
        double min = Double.MAX_VALUE;
        
        synchronized(unitList) {
            for(Unit unit : unitList) {
                if(current.team != unit.team) {
                    double dist = current.getDistance(unit);
                    if(dist < min) {
                        min = dist;
                        gegner = unit;
                    }
                }
            }
        }
        return gegner;
    }
    
    private void updateGameState() {
        List<Unit> toRemove = new ArrayList<>();
        
        synchronized(unitList) {
            for (Unit unit : unitList) {
                Unit gegner = nextGegner(unit);
                
                if (gegner != null && gegner != unit) {
                    double dist = unit.getDistance(gegner);
                    
                    if (dist <= unit.distance) {
                        // Angriff mit stochastischem Bonus
                        double damageBonus = stochasticRandom.nextGaussian() * 2; // ±2 Schaden
                        double actualDamage = Math.max(0, unit.damage + damageBonus);
                        
                        gegner.leben -= actualDamage;
                        gegner.größePerLive(1);
                        
                        if (gegner.leben <= 0 && !toRemove.contains(gegner)) {
                            toRemove.add(gegner);
                        }
                    } else {
                        // Bewegung mit stochastischem Rauschen
                        double noiseX = stochasticRandom.nextGaussian() * 100;
                        double noiseY = stochasticRandom.nextGaussian() * 100;
                        
                        unit.moveTo(gegner);
                        unit.posX += noiseX;
                        unit.posY += noiseY;
                        unit.move(1);
                        unit.größePerLive(1);
                    }
                }
            }
            
            unitList.removeAll(toRemove);
        }
    }

```java
public class MatrixgameCollectiveSimulatorV48 {
    
    // === NEUE SPIEL-MODI ===
    public enum GameMode {
        UNIT_FIGHT,           // Klassisch
        EVE_FACTION_WAR,      // Eve-Online Style
        EMPIRE_AGE,           // Empire Earth
        STARCraft_RACE,       // StarCraft
        SETTLER_ECONOMY,      // Die Siedler
        TAK_GO_HYBRID,        // Brettspiel + Units
        CHESS_STRATEGY        // Schach-ähnlich
    }
    
    // === NEUE UNIT-TYPEN ===
    public enum UnitType {
        // Basis
        INFANTRY, TANK, AIRCRAFT, SHIP,
        
        // Eve-Online
        FRIGATE, CRUISER, BATTLESHIP, CARRIER,
        
        // StarCraft
        MARINE, ZEALOT, ZERGLING, DRAGOON,
        
        // Empire Earth
        CLUBSMAN, KNIGHT, MUSKETEER, TANK_MODERN,
        
        // Spezial
        WORKER, BUILDER, TRADER, RESEARCHER,
        HERO_LEGENDARY, BOSS_EPIC
    }
    
    // === RESSOURCEN-SYSTEM ===
    public static class ResourceManager {
        Map<ResourceType, Double> resources = new HashMap<>();
        
        public enum ResourceType {
            GOLD, WOOD, STONE, FOOD,
            MINERALS, GAS, CRYSTAL,
            ENERGY, POPULATION, KNOWLEDGE
        }
        
        public void gather(Unit worker, ResourceType type, double amount) {
            resources.put(type, resources.getOrDefault(type, 0.0) + amount);
        }
        
        public boolean canAfford(Map<ResourceType, Double> cost) {
            for (Map.Entry<ResourceType, Double> entry : cost.entrySet()) {
                if (resources.getOrDefault(entry.getKey(), 0.0) < entry.getValue()) {
                    return false;
                }
            }
            return true;
        }
    }
    
    // === TECHNOLOGIE-BAUM ===
    public static class TechTree {
        Map<String, Technology> technologies = new HashMap<>();
        
        public class Technology {
            String id;
            String name;
            int era; // 1=Steinzeit, 2=Mittelalter, etc.
            Map<ResourceType, Double> cost;
            List<String> prerequisites;
            double bonusDamage;
            double bonusHP;
            boolean unlocksUnit;
            
            public void research() {
                // Technologie freischalten
                System.out.println("🔬 Forschung abgeschlossen: " + name);
            }
        }
    }
    
    // === TERRITORIUM & SOVEREIGNTY ===
    public static class TerritoryControl {
        Map<Coordinate, Territory> map = new HashMap<>();
        
        public class Territory {
            int ownerTeam;
            double influence;
            List<ResourceNode> resources;
            boolean hasStructure;
            
            public void updateInfluence(int team, double amount) {
                if (ownerTeam == team) {
                    influence += amount;
                } else {
                    influence -= amount;
                    if (influence < 0) {
                        ownerTeam = team;
                        influence = Math.abs(influence);
                        System.out.println("🚩 Territorium erobert von Team " + team);
                    }
                }
            }
        }
    }
    
    // === ALLIANZ & DIPLOMATIE ===
    public static class DiplomacySystem {
        Map<Integer, Set<Integer>> alliances = new HashMap<>();
        Map<Integer, Map<Integer, Relation>> relations = new HashMap<>();
        
        public enum Relation {
            ALLY, NEUTRAL, ENEMY, WAR
        }
        
        public void formAlliance(int team1, int team2) {
            alliances.computeIfAbsent(team1, k -> new HashSet<>()).add(team2);
            alliances.computeIfAbsent(team2, k -> new HashSet<>()).add(team1);
            setRelation(team1, team2, Relation.ALLY);
            System.out.println("🤝 Allianz gegründet: Team " + team1 + " ↔ Team " + team2);
        }
        
        public boolean areAllies(int team1, int team2) {
            return alliances.getOrDefault(team1, new HashSet<>()).contains(team2);
        }
    }
}
```
